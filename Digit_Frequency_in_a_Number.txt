program digitFrequency;
var
    num : longint ; 
    countD: integer ;
    numStr: string ;
    numFirstDigit: char; 
begin

    repeat
    write('give me number :');
    read(num);
    until(num > 0);
    
    
    str(num, numStr);

    repeat
    
        numFirstDigit := numStr[1];
        countD := 0;
        
        while(pos(numFirstDigit,numStr) <> 0) do
        begin
            delete(numStr,pos(numFirstDigit,numStr),1);
            countD := countD + 1;
        end;
        writeln(numFirstDigit,' : ',countD,' time(s)');
        
    until(length(numStr) = 0)

 
 
end.
